    uint8* addr = (uint8*) start_addr;

Servo Motor Controller with PWM & FSM

Designed and implemented a custom VHDL servo controller IP core with 4-state FSM (SWEEP_LEFT, SWEEP_RIGHT, INT_LEFT, INT_RIGHT), PWM signal generation, and Avalon memory-mapped slave interface. Integrated with Nios II processor using interrupt-driven C handlers for real-time angle control via pushbuttons, with 7-segment display feedback showing min/max angles.

VHDL FSM PWM Nios II Avalon Interface C ISR Platform Designer

Key Features

  • 4-state FSM with automatic sweep between configurable min/max angles
  • 20ms PWM period with 1-2ms pulse width control for servo positioning
  • Interrupt-driven communication with Nios II processor
  • Real-time angle adjustment via pushbutton and switch inputs
  • 7-segment display feedback for visual angle monitoring

architecture Behavioral of Lab5 is
1    -- State machine states
    constant SWEEP_LEFT  : std_logic_vector(1 downto 0) := "00";
    constant SWEEP_RIGHT : std_logic_vector(1 downto 0) := "01";
    constant INT_LEFT    : std_logic_vector(1 downto 0) := "10";
    constant INT_RIGHT   : std_logic_vector(1 downto 0) := "11";

    signal state_reg     : std_logic_vector(1 downto 0);
    signal state_next    : std_logic_vector(1 downto 0);
    signal period_counter: unsigned(19 downto 0) := b"00000000000000000000";
    signal angle_count   : unsigned(31 downto 0) := x"0000C350"; -- 50,000
    signal min_angle_reg : unsigned(31 downto 0) := x"0000C350"; -- 50,000
    signal max_angle_reg : unsigned(31 downto 0) := x"000186A0"; -- 100,000

    constant ANGLE_STEP  : unsigned(31 downto 0) := x"000001F4"; -- 500

begin
2    -- State Register
    state_reg_proc: process(clk, reset_n)
    begin
        if reset_n = '0' then
            state_reg <= SWEEP_RIGHT;
        elsif rising_edge(clk) then
            state_reg <= state_next;
        end if;
    end process;

3    -- Next State Logic
    next_state_proc: process(state_reg, angle_count, min_angle_reg, max_angle_reg, write_enable)
    begin
        state_next <= state_reg; -- Default: hold state
        case state_reg is
            when SWEEP_RIGHT =>
                if angle_count >= max_angle_reg then
                    state_next <= INT_RIGHT;
                end if;
            when INT_RIGHT =>
                if write_enable = '1' then
                    state_next <= SWEEP_LEFT;
                end if;
            when SWEEP_LEFT =>
                if angle_count <= min_angle_reg then
                    state_next <= INT_LEFT;
                end if;
            when INT_LEFT =>
                if write_enable = '1' then
                    state_next <= SWEEP_RIGHT;
                end if;
            when others =>
                state_next <= SWEEP_RIGHT;
        end case;
    end process;

4    -- Register Logic: update min/max from Nios II writes
    register_proc: process(clk, reset_n)
    begin
        if reset_n = '0' then
            min_angle_reg <= x"0000C350";
            max_angle_reg <= x"000186A0";
        elsif rising_edge(clk) then
            if write_enable = '1' then
                if write_address = '0' then
                    min_angle_reg <= unsigned(write_data);
                else
                    max_angle_reg <= unsigned(write_data);
                end if;
            end if;
        end if;
    end process;

5    -- 20ms Period Counter (1,000,000 cycles at 50 MHz)
    period_counter_proc: process(clk, reset_n)
    begin
        if reset_n = '0' then
            period_counter <= (others => '0');
        elsif rising_edge(clk) then
            if period_counter >= x"000F4240" then
                period_counter <= (others => '0');
            else
                period_counter <= period_counter + 1;
            end if;
        end if;
    end process;

    -- Angle Counter: increments/decrements by ANGLE_STEP each 20ms period
    angle_counter_proc: process(clk, reset_n)
    begin
        if reset_n = '0' then
            angle_count <= min_angle_reg;
        elsif rising_edge(clk) then
            case state_reg is
                when SWEEP_RIGHT =>
                    if period_counter = 0 then
                        angle_count <= angle_count + ANGLE_STEP;
                    end if;
                when SWEEP_LEFT =>
                    if period_counter = 0 then
                        angle_count <= angle_count - ANGLE_STEP;
                    end if;
                when others =>
                    angle_count <= angle_count;
            end case;
        end if;
    end process;

    -- IRQ: assert when entering interrupt states
    irq_proc: process(clk, reset_n)
    begin
        if reset_n = '0' then
            irq_signal <= '0';
        elsif rising_edge(clk) then
            if state_reg = INT_LEFT or state_reg = INT_RIGHT then
                irq_signal <= '1';
            else
                irq_signal <= '0';
            end if;
        end if;
    end process;

6    -- PWM Output: high when period_counter < angle_count
    pwm_proc: process(clk, reset_n)
    begin
        if reset_n = '0' then
            pwm_out <= '0';
        elsif rising_edge(clk) then
            if period_counter < angle_count then
                pwm_out <= '1';
            else
                pwm_out <= '0';
            end if;
        end if;
    end process;

end Behavioral;

Breathing Rate Detection System

December 2024

Course: Digital Signal Processing

Built a real-time breathing rate detection system on Arduino interfaced with MATLAB. The signal chain runs through an FIR equalizer, then three parallel IIR filters (5th-order Chebyshev lowpass at 12 BPM, 6th-order bandpass 12–40 BPM, 8th-order highpass at 40 BPM). A running standard-deviation tracker on each band's output classifies breathing rate as low, normal, or high, and drives an audio alarm via a speaker.

Arduino MATLAB IIR Filters FIR Equalizer Chebyshev Real-Time DSP Serial I/O

System Architecture

  • FIR equalizer pre-processes the ADC input to flatten the sensor frequency response
  • Three parallel IIR filter banks: Chebyshev LPF (12 BPM), BPF (12–40 BPM), HPF (40 BPM)
  • Running statistics (mean, variance, stdev) computed per band with periodic reset
  • Alarm logic classifies rate as low / normal / high / undetermined and plays tones at 400 Hz, 1000 Hz, or 200 Hz
  • MATLAB handshake over 115200 baud serial - Arduino reads test vectors from MATLAB, sends back 9-column data stream
  • Dither averaging over 160 sub-samples for improved ADC resolution
